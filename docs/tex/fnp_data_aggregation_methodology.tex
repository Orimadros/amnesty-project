\section{Data Source \& Structure}

\begin{frame}{FNP Market Survey Data}
  \textbf{Data Collection}
  \begin{itemize}
    \item FNP conducts market surveys to estimate land prices
    \item Prices reported in \priceunit{} for specific land types
    \item Annual estimates for subset of municipalities $M$
  \end{itemize}

  \vspace{0.6em}
  \textbf{Data Structure}
  \begin{itemize}
    \item Raw data: unique price estimates for each $(m, t, y)$ combination
    \item $m \in M$: municipalities with available data
    \item $t \in T$: land types (e.g., \textit{pastagem formada}, \textit{pastagem nativa})
    \item $y$: year of observation
    \item Many $(m, t, y)$ combinations missing due to data availability
  \end{itemize}
\end{frame}

% ============================================================================
% SECTION: AGGREGATION METHODOLOGY
% ============================================================================
\section{Municipality-Level Aggregation}

\begin{frame}{Land Type Classification}
  \textbf{Macro-categories}
  \begin{itemize}
    \item Land types $T$ grouped into macro-categories (e.g., \textit{pastagem}, \textit{floresta})
    \item Multiple subtypes within each category
  \end{itemize}

  \vspace{0.6em}
  \textbf{Example: Pastagem Category}
  \begin{itemize}
    \item \textit{pastagem formada} (formed pasture)
    \item \textit{pastagem nativa} (native pasture)
  \end{itemize}

  \vspace{0.6em}
  
  \small*Pastagem only has these two varieties in the post-2016 data
\end{frame}

\begin{frame}{Municipality-Level Aggregation Process}
  \textbf{Step 1: Municipality-level aggregation}
  \begin{itemize}
    \item For each municipality $m$ in the 2017 sample:
    \item Identify all land types $t \in T_{\text{pastagem}}$ (pasture subtypes)
    \item Calculate mean price across all pasture subtypes for each year $y$
  \end{itemize}

  \vspace{0.6em}
  \textbf{Final Output}
  \begin{itemize}
    \item 46 Amazon municipalities with pastagem data $\times$ 7 years (2016--2022)
    \item Municipality-level average prices in R\$ for 1ha of pastureland
    \item Each municipality represents a distinct spatial unit
  \end{itemize}
\end{frame}
